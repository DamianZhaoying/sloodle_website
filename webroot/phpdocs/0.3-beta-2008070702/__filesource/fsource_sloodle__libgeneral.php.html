  104.         
  105.         // Attempt to send the data via http
  106.         $response = $client->send(
  107.             $content,
  108.             60,
  109.             'http'
  110.         );

  112.         //var_dump($response); // Debug output

  185.     /**
  186.     * Generates a random login security token.
  187.     * Uses mixed-case letters and numbers to generate a random 16-character string.
  188.     * @return string 
  189.     * @see sloodle_random_web_password()
  190.     */
  191.     function sloodle_random_security_token()
  192.     {
  193.         // Define the characters we can use in our token, and get the length of it
  194.         $str = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
  195.         $strlen = strlen($str) - 1;
  196.         // Prepare the token variable
  197.         $token = '';
  198.         // Loop once for each output character
  199.         for($length = 0; $length < 16; $length++) {
  200.             // Shuffle the string, then pick and store a random character
  201.             $str = str_shuffle($str);
  202.             $char = mt_rand(0, $strlen);
  203.             $token .= $str[$char];
  204.         }
  205.         
  206.         return $token;
  207.     }

  377.         if ( ($posarr['x'] < $minarr['x']) || ($posarr['y'] < $minarr['y']) || ($posarr['z'] < $minarr['z']) ) {

  379.         }

  410.     * @param array $size Associative array giving the size of the zone
  411.     * @return array An associative vector array

  413.     function sloodle_random_position_in_zone($size)
  414.     {
  415.         // Construct the half-size array
  416.         $halfsize = array('x'=>($size['x'] / 2.0) - 2.0, 'y'=>($size['y'] / 2.0) - 2.0, 'z'=>($size['z'] / 2.0) - 2.0);
  417.     
  418.         $pos = array();
  419.         $pos['x'] = mt_rand(0.0, $size['x'] - 4.0) - $halfsize['x'];
  420.         $pos['y'] = mt_rand(0.0, $size['y'] - 4.0) - $halfsize['y'];
  421.         $pos['z'] = mt_rand(0.0, $size['z'] - 4.0) - $halfsize['z'];
  422.         return $pos;
  423.     }

  425.     // Round the specified 3d vector to integer values
  426.     // $pos should be a vector string "<x,y,z>" or an associative array {x,y,z}
  427.     // Return is the same as the type passed-in
  428.     // If the input type is unrecognised, it simply returns it back out unchanged
  429.     /**
  430.     * Rounds the specified 3d vector integer values.
  431.     * Can handle/return a string vector, or an array vector.
  432.     * (Output type matches input type).
  433.     * @param mixed $pos Either a string vector or an array vector
  434.     * @return mixed 
  435.     */
  436.     function sloodle_round_vector($pos)
  437.     {
  438.         // We will work with an array, but allow for conversion to/from string
  439.         $arrayvec = $pos;
  440.         $returnstring = FALSE;
  441.         // Is it a string?
  442.         if (is_string($pos)) {
  443.             $arrayvec = sloodle_vector_to_array($pos);
  444.             $returnstring = TRUE;
  445.         } else if (!is_array($pos)) {
  446.             return $pos;
  447.         }
  448.     
  449.         // Construct an output array
  450.         $output = array();
  451.         foreach ($arrayvec as $key => $val) {
  452.             $output[$key] = round($val, 0);
  453.         }
  454.         
  455.         // If we need to convert it back to a string, then do so
  456.         if ($returnstring) {
  457.             return sloodle_array_to_vector($output);
  458.         }
  459.         
  460.         return $output;
  461.     }

  463.     /**

  466.     * (Or returns false if no LoginZone position/size could be found in the Moodle configuration table).
  467.     * @param string $pos A string vector giving the position of the LoginZone

  471.     function sloodle_login_zone_bounds($pos, $size)

  474.         if (($pos == FALSE) || ($size == FALSE)) {
  475.             return FALSE;
  476.         }
  477.         // Convert both to arrays
  478.         $posarr = sloodle_vector_to_array($pos);

  480.         // Calculate the bounds
  481.         $max = array();
  482.         $max['x'] = $posarr['x']+(($sizearr['x'])/2)-2;

  488.         $min['z'] = $posarr['z']-(($sizearr['z'])/2)+2;

  490.         return array($max,$min);
  491.     }

  494.     /**
  495.     * Checks if the given prim password is valid.
  496.     * @param string $password The password string to check
  497.     * @return bool True if it is valid, or false otherwise.
  498.     */
  499.     function sloodle_validate_prim_password($password)
  500.     {
  501.         // Check that it's a string
  502.         if (!is_string($password)) return false;
  503.         // Check the length
  504.         $len = strlen($password);
  505.         if ($len < 5 || $len > 9) return false;
  506.         // Check that it's all numbers

  508.         // Check that it doesn't start with a 0
  509.         if ($password[0] == '0') return false;

  515.     /**

  518.     * Each error code is a word. The full text of the error message may be obtained

  529.         $result = true;
  530.         
  531.         // Check that it's a string
  532.         if (!is_string($password)) {
  533.             $errors[] = 'invalidtype';
  534.             $result = false;
  535.         }
  536.         // Check the length

  538.         if ($len < 5) {
  539.             $errors[] = 'tooshort';
  540.             $result = false;
  541.         }
  542.         if ($len > 9) {
  543.             $errors[] = 'toolong';
  544.             $result = false;

  555.             $errors[] = 'leadingzero';

  557.         }

  559.         return $result;

  604.     /**
  605.     * Processes pending login notifications, up to a certain limit.
  606.     * Retrieves the requests one-at-a-time for processing.
  607.     * This is slower, but ensures minimal damage if the process is terminated, e.g. due to server timeout.
  608.     * @param int $limit The maximum number of pending requests to process.
  609.     * @return void 
  610.     */
  611.     function sloodle_process_login_notifications($limit = 25)
  612.     {
  613.         global $CFG;
  614.         
  615.         // Validate the limit
  616.         $limit = (int)$limit;
  617.         if ($limit < 1) return;
  618.         
  619.         // Go through each one
  620.         for ($i = 0; $i < $limit; $i++) {
  621.             // Obtain the first record
  622.             $recs = get_records('sloodle_login_notifications', '', '', 'id', '*', 0, $limit);
  623.             if (!$recs) return false;
  624.             reset($recs);
  625.             $rec = current($recs);
  626.             
  627.             // Determine the user ID of the person who requested this
  628.             $userid = 0;
  629.             if (!($sloodleuser = get_record('sloodle_users', 'uuid', $rec->avatar))) {
  630.                 // Failed to the user - get the guest user instead
  631.                 $guestdata = guest_user();
  632.                 $userid = $guestdata->id;
  633.             } else {
  634.                 // Got the data - store the user ID
  635.                 $userid = $sloodleuser->userid;
  636.             }
  637.             
  638.             // Send the notification
  639.             if (sloodle_send_login_notification($rec->destination, $rec->avatar, $rec->username, $rec->password)) {
  640.                 // Log the notification
  641.                  add_to_log(SITEID, 'sloodle', 'view', '', 'Sent login details by email to avatar in-world', 0, $userid);
  642.             } else {
  643.                 // Log the failed notification (but don't keep trying the same one)
  644.                 add_to_log(SITEID, 'sloodle', 'view failed', '', 'Failed to send login details by email to avatar in-world', 0, $userid);
  645.             }
  646.             
  647.             // Delete the record from the data
  648.             delete_records('sloodle_login_notifications', 'id', $rec->id);
  649.         }
  650.     }

  808.     * @param string $dir The directory to search WITHOUT a trailing slash. (Note: cannot search the current directory or higher in the file hierarchy)

  812.     function sloodle_get_subdirectories($dir, $relative = true)
  813.     {

  870.     * Extracts the object name and version number from an object identifier.
  871.     * @param string $objid An object identifier, such as "chat-1.0"
  872.     * @return array A numeric array of name then version number.
  873.     */
  874.     function sloodle_parse_object_identifier($objid)
  875.     {
  876.         // Find the last dash character, and split the string around it.
  877.         $lastpos = strrpos($objid, '-');
  878.         // Check for common problems
  879.         if ($lastpos === false) return array($objid, ''); // No dash
  880.         if ($lastpos == 0) return array('', substr($objid, 1)); // Dash at start
  881.         if ($lastpos == (strlen($objid) - 1)) return array(substr($objid, 0, -1), ''); // Dash at end
  882.         // Split up the values
  883.         $name = substr($objid, 0, $lastpos);
  884.         $version = substr($objid, $lastpos + 1, strlen($objid) - $lastpos - 1);
  885.         return array($name, $version);
  886.     }

  918.             } else {

  921.         }
