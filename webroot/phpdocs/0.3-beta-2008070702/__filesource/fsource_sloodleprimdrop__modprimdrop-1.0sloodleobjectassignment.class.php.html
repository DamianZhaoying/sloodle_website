  253.     /**
  254.     * Parses the data from a Submission database record object.
  255.     * @var object $submission The submission database record object.
  256.     */
  257.     function load_submission($submission)
  258.     {
  259.         // Make sure the submission is valid
  260.         $this->is_loaded = false;
  261.         if ($submission == false || $submission == null) return;
  262.     
  263.         // Parse the data lines
  264.         $object_data = explode('|', $submission->data1);
  265.         $primdrop_data = explode('|', $submission->data2);
  266.         
  267.         // Make sure there's enough data in both
  268.         if (count($object_data) < 2 || count($primdrop_data) < 4) return;
  269.         
  270.         // Extract the data items
  271.         $this->obj_name = $object_data[0];
  272.         $this->num_prims = (int)$object_data[1];
  273.         $this->primdrop_name = $primdrop_data[0];
  274.         $this->primdrop_uuid = $primdrop_data[1];
  275.         $this->primdrop_region = $primdrop_data[2];
  276.         $this->primdrop_pos = $primdrop_data[3];
  277.         
  278.         $this->is_loaded = true;
  279.     }

  326.     * Construct a text summary of this submission.

  340.             $arr = sloodle_vector_to_array($this->primdrop_pos);

  348.             $loc = "secondlife://{$this->primdrop_region}/{$arr['x']}/{$arr['y']}/{$arr['z']}";
  349.             $text .= "<b><a href=\"$loc\">$loc</a></b><br>";
  350.         }
  351.         
  352.         if ($return) return $text;
  353.         echo $text;

  355.     }
