  36.     } else if (empty($CFG->smtphosts)) {

  39.     } else {

  41.         if (!empty($CFG->debugsmtp)) {
  42.             echo '<pre>' . "\n";

  73.         if (!empty($CFG->debugsmtp)) {
  74.             echo '</pre>';

  77.         
  78.     } else {
  79.         mtrace('ERROR: '. $mail->ErrorInfo);

  81.         if (!empty($CFG->debugsmtp)) {
  82.             echo '</pre>';
  83.         }
  84.         return false;
  85.     }
  86. }
