  88.     * Checks for incoming request data identifying a particular module (parameter 'id').
  89.     * Loads the relevant course module and SLOODLE module instances.
  90.     * This should be overridden to add functionality to load any module-specific data.

  92.     */
  93.     function process_request()
  94.     {
  95.         // Note: some modules prefer 's' to indicate the instance number... may need to implement that as well.
  96.         // Fetch the course module instance
  97.         $id = required_param('id', PARAM_INT);
  98.         if (!$this->cm = get_coursemodule_from_id('sloodle', $id)) error('Course module ID was incorrect.');
  99.         // Fetch the course data
  100.         if (!$this->course = get_record('course', 'id', $this->cm->course)) error('Failed to retrieve course.');
  101.         $this->sloodle_course = new SloodleCourse();
  102.         if (!$this->sloodle_course->load($this->course)) error(get_string('failedcourseload', 'sloodle'));
  103.  
  104.         // Fetch the SLOODLE instance itself
  105.         if (!$this->sloodle = get_record('sloodle', 'id', $this->cm->instance)) error('Failed to find SLOODLE module instance');
  106.     }

  110.     */

  112.     {
  113.         // Make sure the user is logged-in
  114.         require_course_login($this->course, true, $this->cm);
  115.  
  116.         add_to_log($this->course->id, 'sloodle', 'view sloodle module', "view.php?id={$this->cm->id}", "{$this->sloodle->id}", $this->cm->id);
  117.         
  118.         // Check for permissions
  119.         $this->module_context = get_context_instance(CONTEXT_MODULE, $this->cm->id);
  120.         $this->course_context = get_context_instance(CONTEXT_COURSE, $this->course->id);
  121.         if (has_capability('moodle/course:manageactivities', $this->module_context)) $this->canedit = true;
  122.  
  123.         // If the module is hidden, then can the user still view it?
  124.         if (empty($this->cm->visible) && !has_capability('moodle/course:viewhiddenactivities', $this->module_context)) notice(get_string('activityiscurrentlyhidden'));
  125.     }

  153.     
  154.         // Display the module type and description
