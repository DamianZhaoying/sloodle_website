  58.     function process_request()
  59.     {

  61.         if (!$this->course = get_record('course', 'id', $id)) error('Could not find course.');

  69.     function check_permission()
  70.     {
  71.         // Ensure the user logs in
  72.         require_login($this->course->id);
  73.         if (isguestuser()) error(get_string('noguestaccess', 'sloodle'));
  74.         add_to_log($this->course->id, 'course', 'view sloodle data', '', "{$this->course->id}");
  75.  
  76.         // Ensure the user is allowed to update information on this course
  77.         $this->course_context = get_context_instance(CONTEXT_COURSE, $this->course->id);
  78.         require_capability('moodle/course:update', $this->course_context);
  79.     }
