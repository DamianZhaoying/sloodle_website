  17.     
  18.     /*
  19.     * The following parameter is always required:
  20.     *
  21.     *  sloodleauthid = the ID of the entry being authorised
  22.     *
  23.     * If an avatar is being registered at the same time, then both of the following parameters are also required:
  24.     *

  26.     *  sloodlelst = the system-generated login security token
  27.     *
  28.     */
  29.     
  30.     /** Include Sloodle/Moodle configuration. */
  31.     require_once('../sl_config.php');

  33.

  35.     sloodle_require_login_no_guest();
  36.     
  37.     /** Include the Sloodle API. */
  38.     require_once(SLOODLE_LIBROOT.'/sloodle_session.php');
  39.     
  40.     // Display the page header
  41.     $strsloodle = get_string('modulename', 'sloodle');
  42.     print_header_simple(get_string('userobjectauth', 'sloodle'), "", get_string('userobjectauth', 'sloodle'), "", "", true);
  43.     
  44.     // Make sure it's not a guest who is logged in
  45.     if (isguest()) {
  46.         ?>
  47.         <div style="text-align:center;">
  48.          <h3><?php print_string('error', 'sloodle'); ?></h3>
  49.          <p><?php print_string('noguestaccess', 'sloodle'); ?></p>
  50.         </div>
  51.         <?php
  52.         print_footer();
  53.         exit();
  54.     }
  55.     
  56.     // Process the request data
  57.     $sloodle = new SloodleSession();
  58.     // Load the Moodle user and linked avatar
  59.     $sloodle->user->load_user($USER->id);
  60.     $sloodle->user->load_linked_avatar();
  61.     
  62.     // Get the authorisation ID

  66.     if (!$sloodle->user->is_avatar_loaded()) {
  67.         // Make sure the user has permission to register their avatar

  69.     
  70.         // Get the parameters

  72.         $sloodlelst = required_param('sloodlelst', PARAM_TEXT);
  73.         
  74.         // Attempt to find a pending avatar entry which matches the given details
  75.         $pa = get_record('sloodle_pending_avatars', 'uuid', $sloodleuuid, 'lst', $sloodlelst);
  76.         if (!$pa) {
  77.             ?>
  78.             <div style="text-align:center;">
  79.              <h3><?php print_string('error', 'sloodle'); ?></h3>

  81.             </div>
  82.             <?php
  83.             print_footer();
  84.             exit();
  85.         }
  86.         
  87.         // Add the new avatar
  88.         if (!$sloodle->user->add_linked_avatar($USER->id, $sloodleuuid, $pa->avname)) {
  89.             // Failed
  90.             ?>
  91.             <div style="text-align:center;">
  92.              <h3><?php print_string('error', 'sloodle'); ?></h3>
  93.              <p><?php print_string('failedcreatesloodleuser', 'sloodle'); ?></p>
  94.             </div>
  95.             <?php
  96.             print_footer();
  97.             exit();
  98.         }
  99.     }

  101.     // Make sure the user has permission to authorise user objects

  103.  
  104.     // Attempt to authorise the object

  106.         ?>
  107.         <div style="text-align:center;">
  108.          <p><?php print_string('objectauthsuccessful', 'sloodle'); ?></p>
  109.         </div>
  110.         <?php
  111.     } else {
  112.         ?>
  113.         <div style="text-align:center;">
  114.          <h3><?php print_string('error', 'sloodle'); ?></h3>
  115.          <p><?php print_string('objectauthfailed', 'sloodle'); ?></p>
  116.         </div>
  117.         <?php
  118.     }
  119.     
  120.     
  121.     print_footer();
