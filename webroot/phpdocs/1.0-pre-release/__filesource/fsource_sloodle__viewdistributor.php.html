  58.     /**
  59.     * Render the view of the Distributor.
  60.     */
  61.     function render()
  62.     {
  63.         global $CFG, $USER;
  64.     
  65.         // Fetch a list of all distributor entries
  66.         $entries = get_records('sloodle_distributor_entry', 'distributorid', $this->distributor->id, 'name');
  67.         // If the query failed, then assume there were simply no items available
  68.         if (!is_array($entries)) $entries = array();
  69.         $numitems = count($entries);
  70.         
  71.         
  72.         // A particular default user can be requested (by avatar name) in the HTTP parameters.
  73.         // This could be used with a "send to this avatar" button on a Sloodle user profile.
  74.         $defaultavatar = optional_param('defaultavatar', null, PARAM_TEXT);
  75.         
  76.         
  77.         // // SEND OBJECT // //
  78.         
  79.         // If the user and object parameters are set, then try to send an object
  80.         if (isset($_REQUEST['user'])) $send_user = $_REQUEST['user'];
  81.         if (isset($_REQUEST['object'])) $send_object = $_REQUEST['object'];
  82.         if (!empty($send_user) && !empty($send_object)) {
  83.  
  84.             // Convert the HTML entities back again
  85.             $send_object = htmlentities(stripslashes($send_object));
  86.  
  87.             // Construct and send the request
  88.             $request = "1|OK\\nSENDOBJECT|$send_user|$send_object";
  89.             $ok = sloodle_send_xmlrpc_message($this->distributor->channel, 0, $request);
  90.             
  91.             // What was the result?
  92.             print_box_start('generalbox boxaligncenter boxwidthnarrow centerpara');
  93.             if ($ok) {
  94.                 print '<h3 style="color:green;text-align:center;">'.get_string('sloodleobjectdistributor:successful','sloodle').'</h3>';
  95.             } else {
  96.                 print '<h3 style="color:red;text-align:center;">'.get_string('sloodleobjectdistributor:failed','sloodle').'</h3>';
  97.             }
  98.             print '<p style="text-align:center;">';
  99.                 print get_string('Object','sloodle').': '.$send_object.'<br/>';
  100.                 print get_string('uuid','sloodle').': '.$send_user.'<br/>';
  101.                 print get_string('xmlrpc:channel','sloodle').': '.$this->distributor->channel.'<br/>';
  102.                 print '</p>';
  103.             print_box_end();
  104.         }
  105.         
  106.         // // ----------- // //
  107.         
  108.  
  109.         // If there are no items in the distributor, then simply display an error message
  110.         if ($numitems < 1) print_box('<span style="font-weight:bold; color:red;">'.get_string('sloodleobjectdistributor:noobjects','sloodle').'</span>', 'generalbox boxaligncenter boxwidthnormal centerpara');
  111.         //error(get_string('sloodleobjectdistributor:noobjects','sloodle'));
  112.         // If there is no XMLRPC channel specified, then display a warning message
  113.         $disabledattr = '';
  114.         if (empty($this->distributor->channel)) {
  115.             print_box('<span style="font-weight:bold; color:red;">'.get_string('sloodleobjectdistributor:nochannel','sloodle').'</span>', 'generalbox boxaligncenter boxwidthnormal centerpara');
  116.             $disabledattr = 'disabled="true"';
  117.         }
  118.         
  119.         // Construct the selection box of items
  120.         $selection_items = '<select name="object" size="1">';
  121.         foreach ($entries as $e) {
  122.             $escapedname = stripslashes($e->name);
  123.             $selection_items .= "<option value=\"{$e->name}\">{$escapedname}</option>\n";
  124.         }
  125.         $selection_items .= '</select>';
  126.         
  127.         // Get a list of all avatars on the site
  128.         $avatars = get_records('sloodle_users', '', '', 'avname');
  129.         if (!$avatars) $avatars = array();
  130.         // Construct the selection box of avatars
  131.         $selection_avatars = '<select name="user" size="1">';
  132.         foreach ($avatars as $a) {
  133.             // Skip avatars who do not have a UUID or associated Moodle account
  134.             if (empty($a->uuid) || empty($a->userid)) continue;
  135.             // Make sure the associated Moodle user can view the current course
  136.             if (!has_capability('moodle/course:view', $this->course_context, $a->userid)) continue;
  137.             // Make sure the associated Moodle user does not have a guest role
  138.             if (has_capability('moodle/legacy:guest', $this->course_context, $a->userid, false)) continue;
  139.  
  140.             $sel = '';
  141.             if ($a->avname == $defaultavatar) $sel = 'selected="true"';
  142.             $selection_avatars .= "<option value=\"{$a->uuid}\" $sel>{$a->avname}</option>\n";
  143.         }
  144.         $selection_avatars .= '</select>';
  145.         
  146.  
  147.         // There will be 3 forms:
  148.         //  - send to self
  149.         //  - send to another avatar on the course
  150.         //  - send to custom UUID
  151.         // The first 1 will be available to any registered user whose avatar is in the database.
  152.         // The other 2 will only be available to those with the activity management capability.
  153.         // Furthermore, the 2nd form will only be available if there is at least 1 avatar registered on the site.
  154.  
  155.         // Start of the sending forms
  156.         print_box_start('generalbox boxaligncenter boxwidthnormal centerpara');
  157.         
  158.     // // SEND TO SELF // //
  159.         
  160.         // Start the form
  161.         echo '<form action="" method="POST">';
  162.         
  163.         // Use a table for layout
  164.         $table_sendtoself = new stdClass();
  165.         $table_sendtoself->head = array(get_string('sloodleobjectdistributor:sendtomyavatar','sloodle'));
  166.         $table_sendtoself->align = array('center');
  167.         
  168.         // Fetch the current user's Sloodle info
  169.         $this->sloodleuser = get_record('sloodle_users', 'userid', $USER->id);
  170.         if (!$this->sloodleuser) {
  171.             $table_sendtoself->data[] = array('<span style="color:red;">'.get_string('avatarnotlinked','sloodle').'</span>');
  172.         } else {
  173.             // Output the hidden form data
  174.             echo <<<XXXEODXXX
  175.  <input type="hidden" name="s" value="{$this->sloodle->id}">
  176.  <input type="hidden" name="user" value="{$this->sloodleuser->uuid}">
  177. XXXEODXXX;
  178.         
  179.             // Object selection box
  180.             $table_sendtoself->data[] = array(get_string('selectobject','sloodle').': '.$selection_items);
  181.             // Submit button
  182.             $table_sendtoself->data[] = array('<input type="submit" '.$disabledattr.' value="'.get_string('sloodleobjectdistributor:sendtomyavatar','sloodle').' ('.$this->sloodleuser->avname.')" />');
  183.         }
  184.         
  185.         // Print the table
  186.         print_table($table_sendtoself);
  187.         
  188.         // End the form
  189.         echo "</form>";
  190.         
  191.         
  192.         // Only show the other options if the user has permission to edit stuff
  193.         if ($this->canedit) {
  194.         // // SEND TO ANOTHER AVATAR // //
  195.             
  196.             // Start the form
  197.             echo '<br><form action="" method="POST">';
  198.             
  199.             // Use a table for layout
  200.             $table = new stdClass();
  201.             $table->head = array(get_string('sloodleobjectdistributor:sendtoanotheravatar','sloodle'));
  202.             $table->align = array('center');
  203.             
  204.             // Do we have any avatars?
  205.             if (count($avatars) < 1) {
  206.                 $table->data[] = array('<span style="color:red;">'.get_string('nosloodleusers','sloodle').'</span>');
  207.             } else {
  208.                 // Output the hidden form data
  209.                 echo <<<XXXEODXXX
  210.      <input type="hidden" name="s" value="{$this->sloodle->id}">
  211. XXXEODXXX;
  212.                 // Avatar selection box
  213.                 $table->data[] = array(get_string('selectuser','sloodle').': '.$selection_avatars);
  214.                 // Object selection box
  215.                 $table->data[] = array(get_string('selectobject','sloodle').': '.$selection_items);
  216.                 // Submit button
  217.                 $table->data[] = array('<input type="submit" '.$disabledattr.' value="'.get_string('sloodleobjectdistributor:sendtoanotheravatar','sloodle').'" />');
  218.             }
  219.             
  220.             // Print the table
  221.             print_table($table);
  222.             
  223.             // End the form
  224.             echo "</form>";
  225.             
  226.         // // SEND TO A CUSTOM AVATAR // //
  227.             
  228.             // Start the form

  230.             
  231.             // Use a table for layout

  234.             $table->align = array('center');
  235.             
  236.             // Output the hidden form data
  237.             echo <<<XXXEODXXX
  238. <input type="hidden" name="s" value="{$this->sloodle->id}">
  239. XXXEODXXX;
  240.         
  241.             // UUID box
  242.             $table->data[] = array(get_string('uuid','sloodle').': '.'<input type="text" name="user" size="46" maxlength="36" />');
  243.             // Object selection box
  244.             $table->data[] = array(get_string('selectobject','sloodle').': '.$selection_items);
  245.             // Submit button
  246.             $table->data[] = array('<input type="submit" '.$disabledattr.' value="'.get_string('sloodleobjectdistributor:sendtocustomavatar','sloodle').'" />');
  247.             
  248.             // Print the table
  249.             print_table($table);
  250.             
  251.             // End the form
  252.             echo "</form>";
