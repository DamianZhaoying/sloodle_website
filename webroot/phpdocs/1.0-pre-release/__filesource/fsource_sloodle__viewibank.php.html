  232.     function process_form()
  233.     {        
  234.         //STEP 1 - HAS THE DEFAULT STIPEND AMOUNT BEEN ALLOCATED TO EVERY USER?       
  235.         //======================== creates initial default stipend records for all users

  238.         if (!$this->stipendGiverObj->getTransactionRecords()) {
  239.               //create stipend transactions for each student!
  240.               //get student list
  241.                
  242.                //for each user in the class insert a transaction record with following values:
  243.                //avuuid,userid,avname,amount,type,timemodified
  244.              
  245.              foreach ($this->sCourseObj->getUserList() as $u){ 
  246.              //Ipoints can be various types - ie: real lindens, or just points.                         
  247.                     $iTransaction = new stdClass();
  248.                     $iTransaction->userid       = $u->id;
  249.                     $iTransaction->sloodleid    = $this->stipendGiverObj->getSloodleId();                 
  250.                     $iTransaction->icurrency     = $this->stipendGiverObj->geticurrency();
  251.                     $iTransaction->amount       = $this->stipendGiverObj->getStartingBalance();
  252.                     $iTransaction->itype = "stipend";
  253.                     $iTransaction->timemodified = time();
  254.                     $this->stipendGiverObj->makeTransaction($iTransaction);
  255.              }    
  256.            //insert iTransaction into db            
  257.            }
  258.               
  259.         
  260.        //STEP 2 - Transactions exist for this stipend already, check to see if new students have been added
  261.        //========================= HAVE NEW STUDENTS BEEN ADDED 
  262.        else 
  263.        {
  264.            $this->stipendGiverObj->updateUserList($this->sCourseObj->getUserList());
  265.         }
  266.         //============== STEP 3 - CHECK TO SEE IF THERE ARE PENDING UPDATES TO EXISTING STUDENTS
  267.            //check if we need to update allocations 
  268.            //if "update" param exists, that means someone submitted a form using the update butotn in printUserTable function
  269.            $update= optional_param("update");
  270.             if ($update){
  271.                 
  272.                 //check each allotment value from form, and compare against db. change where necessary
  273.                 $newBudgets=optional_param("newbudgets");
  274.                 $userIds = optional_param("userIds");
  275.                 $userNames = optional_param("usernames");
  276.                 $currIndex=0;   
  277.                 $updatedRecs = Array();
  278.                 $errorString=''; 
  279.                 foreach ($userIds as $userId) {
  280.                     //retreive saved budget for this user from the database
  281.                     $savedBudget = $this->stipendGiverObj->getStipendBudget($userId,$this->userList);                    
  282.                     //compare saved budget with the budget submitted in the form field for this user
  283.                    
  284.                     if ((int)$newBudgets[$currIndex] !==$savedBudget){
  285.                         //The user has submited a new stipend allotment for this user, because the new budget 
  286.                         //from the form differs from what has been saved in the db.
  287.                         //so we must update the alloted amount in db
  288.                         //STEP 1 First we should check if the user has already withdrawn any stipends
  289.                         $withdrawnAmount = 0;
  290.                         $withdrawnAmount = $this->stipendGiverObj->getUserDebits($userId,$this->userList);
  291.                         if ($withdrawnAmount > $newBudgets[$currIndex]){
  292.                             $errorString = "<br>".$userNames[$currIndex]." ".get_string('stipendgiver:alreadywd','stipendgiver') . $withdrawnAmount;
  293.                             $errorString =  get_string('stipendgiver:alreadywd2','stipendgiver');
  294.                         }else{
  295.                                 //STEP 2 - get id of transaction to update
  296.                                 $stipendRec = $this->stipendGiverObj->getStipendRec($userId);
  297.                                
  298.                                 $transactionUpdate = new stdClass();
  299.                                 $transactionUpdate->id=$stipendRec->id;
  300.                                 $transactionUpdate->amount=$newBudgets[$currIndex];                            
  301.                                 $transactionUpdate->timemodified= time();
  302.                                 $this->stipendGiverObj->updateTransaction($transactionUpdate);
  303.                                 $updatedRecs[]=$userNames[$currIndex];
  304.                                 
  305.                         }
  306.                     }
  307.                     $currIndex++;     
  308.                     
  309.                 }
  310.                 //create and print confirmation message to the user
  311.                 $confirmedMessage = get_string("stipendgiver:successfullupdate","sloodle");
  312.                 $confirmedMessage .= $this->addCommas($updatedRecs);
  313.                 
  314.                 //send confirmation Message
  315.                 $this->setRenderMessage($confirmedMessage . $errorString);
  316.                 //updater stipendgier ibank
  317.                 $this->stipendGiverObj->update();
  318.             }
  319.        
  320.        }
