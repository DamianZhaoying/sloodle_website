  6. * @package sloodle

  9.  
  10. /** Include the current Sloodle configuration, if possible. */
  11. @include_once($CFG->dirroot .'/mod/sloodle/sl_config.php');
  12. if (defined('SLOODLE_LIBROOT')) {
  13.     /** Inlcude the current general Sloodle functionality. */
  14.     require_once(SLOODLE_LIBROOT.'/general.php');
  15.     /** Include the Sloodle course functionality. */
  16.     require_once(SLOODLE_LIBROOT.'/course.php');
  17. }
  18.  
  19. /** Include the old Sloodle configuration, in case the module is out-dated. */
  20. if (!defined('SLOODLE_VERSION')) {
  21.     @include_once($CFG->dirroot .'/mod/sloodle/config.php');
  22.     if (defined('SLOODLE_DIRROOT')) {
  23.         /** Include the old general Sloodle functionality. */
  24.         require_once(SLOODLE_DIRROOT.'/lib/sl_generallib.php');
  25.     }
  26. }
  27.  
  28. /** Define the Sloodle Menu Block version. */
  29. define('SLOODLE_MENU_VERSION', 0.3);
  30.  
  31. /**
  32. * Defines the block class.
  33. * @package sloodle
  34. */
  35. class block_sloodle_menu extends block_base {
  36.  
  37.     /**
  38.     * Perform block initialisation.
  39.     * @return void 
  40.     */
  41.     function init() {
  42.         global $CFG;
  43.         
  44.         $this->title = get_string('blockname', 'block_sloodle_menu');
  45.         $this->content_type = BLOCK_TYPE_TEXT;
  46.         $this->version = 2008060200;
  47.     }
  48.     
  49.     /**
  50.     * Indicates whether or not this module has a global configuration page.
  51.     * @return bool True if there is a global configuration page, or false otherwise.
  52.     */
  53.     function has_config() {
  54.         return false;
  55.     }
  56.     
  57.     /**
  58.     * Indicates whether or not to hide the header of this block.
  59.     * @return bool True to hide the header, or false to show it.
  60.     */
  61.     function hide_header() {
  62.         return false;
  63.     }
  64.  
  65.     /**
  66.     * Defines *and* returns the content of this block.
  67.     * @return object 
  68.     */
  69.     function get_content() {
  70.         global $CFG, $COURSE, $USER;
  71.         
  72.         // Construct the content
  73.         $this->content = new stdClass;
  74.         $this->content->text = '';
  75.         $this->content->footer = '';
  76.         
  77.         // If no course has been specified, then we are using the site course
  78.         if (!isset($COURSE)) {
  79.             $COURSE = get_site();
  80.         }
  81.         
  82.         // If the user is not logged in or if they are using guest access, then we can't show anything
  83.         if (!isloggedin() || isguest()) {
  84.             return $this->content;
  85.         }
  86.         
  87.         // Get the context instance for this course
  88.         $course_context = get_context_instance(CONTEXT_COURSE, $COURSE->id);
  89.         
  90.         // This version of the menu isn't compatible with older version of the Sloodle module
  91.         if (SLOODLE_VERSION < 0.3) {
  92.             $this->content->text = get_string('oldmodule', 'block_sloodle_menu');
  93.             return $this->content;
  94.         }
  95.         
  96.         // Has the Sloodle activity module been installed?
  97.         if (!(function_exists("sloodle_is_installed") && sloodle_is_installed())) {
  98.             $this->content->text = get_string('sloodlenotinstalled', 'block_sloodle_menu');
  99.             return $this->content;
  100.         }
  101.         
  102.         // Get the Sloodle course data
  103.         $sloodle_course = new SloodleCourse();
  104.         if (!$sloodle_course->load((int)$COURSE->id)) {
  105.             $this->content->text = get_string('failedloadcourse', 'block_sloodle_menu');
  106.             return $this->content;
  107.         }
  108.         
  109.         // Add the Sloodle and Sloodle Menu version info to the footer of the block
  110.         $this->content->footer = '<span style="color:#565656;font-style:italic; font-size:10pt;">'.get_string('sloodlemenuversion', 'block_sloodle_menu').': '.(string)SLOODLE_MENU_VERSION.'</span>';
  111.         $this->content->footer .= '<br/><span style="color:#888888;font-style:italic;font-size:8pt;">'.get_string('sloodleversion', 'block_sloodle_menu').': '.(string)SLOODLE_VERSION.'</span>';
  112.         
  113.         // Attempt to find a Sloodle user for the Moodle user
  114.         $dbquery = "    SELECT * FROM {$CFG->prefix}sloodle_users
  115.                         WHERE userid = {$USER->id} AND NOT (avname = '' AND uuid = '')
  116.                     ";
  117.         $dbresult = get_records_sql($dbquery);
  118.         $sl_avatar_name = "";
  119.         if (!is_array($dbresult) || count($dbresult) == 0) $userresult = FALSE;
  120.         else if (count($dbresult) > 1) $userresult = "Multiple avatars associated with your Moodle account.";
  121.         else {
  122.             $userresult = TRUE;
  123.             reset($dbresult);
  124.             $cur = current($dbresult);
  125.             $sl_avatar_name = $cur->avname;
  126.         }
  127.         
  128.         if ($userresult === TRUE) {
  129.             // Success
  130.             // Make sure there was a name
  131.             if (empty($sl_avatar_name)) $sl_avatar_name = '('.get_string('nameunknown', 'block_sloodle_menu').')';
  132.             $this->content->text .= '<center><span style="font-size:10pt;font-style:italic;color:#777777;">'.get_string('youravatar', 'block_sloodle_menu').':</span><br/>';
  133.             
  134.             // Make the avatar name a link if the user management page exists
  135.             $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/view/view_user.php?id={$USER->id}&amp;course={$COURSE->id}\">$sl_avatar_name</a>";
  136.             $this->content->text .= '<br/></center>';
  137.             
  138.         } else if (is_string($userresult)) {
  139.             // An error occurred
  140.             $this->content->text .= '<center><span style="font-size:10pt;font-style:italic;color:#777777;">'.get_string('youravatar', 'block_sloodle_menu').':</span><br/>ERROR ('.$userresult.')</center>';
  141.             
  142.         } else {
  143.             // No avatar linked yet
  144.             $this->content->text .= '<center><span style="font-style:italic;">('.get_string('noavatar', 'block_sloodle_menu').')</span></center>';
  145.         }
  146.         
  147.         // Add links to common Sloodle stuff
  148.         $this->content->text .= '<div style="padding:1px; margin-top:4px; margin-bottom:4px; border-top:solid 1px #cccccc; border-bottom:solid 1px #cccccc;">';
  149.  
  150.         // Add the Sloodle profile link
  151.         $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/user.gif\" width=\"16\" height=\"16\"/> ";
  152.         $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/view/view_user.php?id={$USER->id}&amp;course={$COURSE->id}\">".get_string('mysloodleprofile', 'block_sloodle_menu')."</a><br/>";
  153.  
  154.         // Show a link to all Sloodle activities on this course
  155.         //TODO: possibly show number of visible Sloodle activities?
  156.         $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/boxes.gif\" width=\"16\" height=\"16\"/> ";
  157.         $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/index.php?id={$COURSE->id}\">".get_string('sloodleactivities', 'block_sloodle_menu')."</a><br/>";
  158.  
  159.         // Show a link to the LoginZone for this course
  160.         $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/loginzone.gif\" width=\"16\" height=\"16\"/> ";
  161.         // Do we have LoginZone data for this course?
  162.         if ($sloodle_course->has_loginzone_data()) {
  163.             $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/classroom/loginzone.php?id={$COURSE->id}\">".get_string('courseloginzone', 'block_sloodle_menu')."</a><br/>";
  164.         } else {
  165.             $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/classroom/loginzone.php?id={$COURSE->id}\" style=\"color:red; text-decoration:line-through;\" title=\"".get_string('courseloginzone:nodata','block_sloodle_menu')."\">".get_string('courseloginzone', 'block_sloodle_menu')."</a><br/>";
  166.         }
  167.         
  168.         $this->content->text .= '<hr>';
  169.         
  170.         // Add a user management link if the user can view hidden user details
  171.         if (has_capability('moodle/user:viewhiddendetails', $course_context)) {
  172.             $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/user_mng.gif\" width=\"16\" height=\"16\"/> ";
  173.             $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/view/view_users.php?course={$COURSE->id}\">".get_string('usermanagement', 'block_sloodle_menu')."</a><br/>";            
  174.         }
  175.         
  176.         // Add a link to Sloodle course settings, if the user can update the course
  177.         if (has_capability('moodle/course:update', $course_context)) {
  178.             $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/page.gif\" width=\"16\" height=\"16\"/> ";
  179.             $this->content->text .= "<a href=\"{$CFG->wwwroot}/mod/sloodle/view/view_course.php?id={$COURSE->id}\">".get_string('editcourse', 'block_sloodle_menu')."</a><br>\n";
  180.         }
  181.         
  182.         // Add a module configuration link if the user has authority to administer the module
  183.         if (has_capability('moodle/site:config', $course_context)) {
  184.             
  185.             // The address of the configuration page depends on our version of Moodle
  186.             if ($CFG->version < 2007101500) {
  187.                 // < 1.9
  188.                 $address = "/admin/module.php?module=sloodle";
  189.             } else {
  190.                 // >= 1.9
  191.                 $address = "/admin/settings.php?section=modsettingsloodle";
  192.             }
  193.         
  194.             $this->content->text .= "<img src=\"{$CFG->wwwroot}/blocks/sloodle_menu/img/configure.gif\" width=\"16\" height=\"16\"/> ";
  195.             $this->content->text .= "<a href=\"{$CFG->wwwroot}$address\">".get_string('sloodleconfig', 'block_sloodle_menu')."</a><br/>";
  196.         }
  197.         
  198.         $this->content->text .= '</div>';
  199.         
  200.         return $this->content;
  201.     }
  202. }
