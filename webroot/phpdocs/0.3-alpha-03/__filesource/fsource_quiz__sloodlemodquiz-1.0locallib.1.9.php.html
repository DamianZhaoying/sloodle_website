  52.  * Creates an attempt object to represent an attempt at the quiz by the current
  53.  * user starting at the current time. The ->id field is not set. The object is
  54.  * NOT written to the database.
  55.  * @return object                The newly created attempt object.
  56.  * @param object $quiz           The quiz to create an attempt for.

  59. function quiz_create_attempt($quiz, $attemptnumber) {
  60.     global $USER, $CFG;
  61.  
  62.     if (!$attemptnumber > 1 or !$quiz->attemptonlast or !$attempt = get_record('quiz_attempts', 'quiz', $quiz->id, 'userid', $USER->id, 'attempt', $attemptnumber-1)) {
  63.         // we are not building on last attempt so create a new attempt
  64.         $attempt->quiz = $quiz->id;
  65.         $attempt->userid = $USER->id;
  66.         $attempt->preview = 0;
  67.         if ($quiz->shufflequestions) {
  68.             $attempt->layout = quiz_repaginate($quiz->questions, $quiz->questionsperpage, true);
  69.         } else {
  70.             $attempt->layout = $quiz->questions;
  71.         }
  72.     }
  73.  
  74.     $timenow = time();
  75.     $attempt->attempt = $attemptnumber;
  76.     $attempt->sumgrades = 0.0;
  77.     $attempt->timestart = $timenow;
  78.     $attempt->timefinish = 0;
  79.     $attempt->timemodified = $timenow;
  80.     $attempt->uniqueid = question_new_attempt_uniqueid();
  81.  
  82.     return $attempt;
  83. }

  106. function quiz_delete_attempt($attempt, $quiz) {
  107.     if (is_numeric($attempt)) {
  108.         if (!$attempt = get_record('quiz_attempts', 'id', $attempt)) {
  109.             return;
  110.         }
  111.     }
  112.  
  113.     if ($attempt->quiz != $quiz->id) {
  114.         debugging("Trying to delete attempt $attempt->id which belongs to quiz $attempt->quiz " .
  115.                 "but was passed quiz $quiz->id.");
  116.         return;
  117.     }
  118.  
  119.     delete_records('quiz_attempts', 'id', $attempt->id);
  120.     delete_attempt($attempt->uniqueid);
  121.  
  122.     // Search quiz_attempts for other instances by this user.
  123.     // If none, then delete record for this quiz, this user from quiz_grades
  124.     // else recalculate best grade
  125.  
  126.     $userid = $attempt->userid;
  127.     if (!record_exists('quiz_attempts', 'userid', $userid, 'quiz', $quiz->id)) {
  128.         delete_records('quiz_grades', 'userid', $userid,'quiz', $quiz->id);
  129.     } else {
  130.         quiz_save_best_grade($quiz, $userid);

  195. /**
  196.  * Re-paginates the quiz layout
  197.  *
  198.  * @return string         The new layout string
  199.  * @param string $layout  The string representing the quiz layout.
  200.  * @param integer $perpage The number of questions per page
  201.  * @param boolean $shuffle Should the questions be reordered randomly?
  202.  */
  203. function quiz_repaginate($layout, $perpage, $shuffle=false) {
  204.     $layout = str_replace(',0', '', $layout); // remove existing page breaks
  205.     $questions = explode(',', $layout);
  206.     if ($shuffle) {
  207.         srand((float)microtime() * 1000000); // for php < 4.2
  208.         shuffle($questions);
  209.     }
  210.     $i = 1;
  211.     $layout = '';
  212.     foreach ($questions as $question) {
  213.         if ($perpage and $i > $perpage) {
  214.             $layout .= '0,';
  215.             $i = 1;
  216.         }
  217.         $layout .= $question.',';
  218.         $i++;
  219.     }
  220.     return $layout.'0';
  221. }

  258. /**
  259.  * Creates an array of maximum grades for a quiz

  261.  * The grades are extracted from the quiz_question_instances table.
  262.  * @return array        Array of grades indexed by question id
  263.  *                       These are the maximum possible grades that
  264.  *                       students can achieve for each of the questions

  267. function quiz_get_all_question_grades($quiz) {
  268.     global $CFG;
  269.  
  270.     $questionlist = quiz_questions_in_quiz($quiz->questions);
  271.     if (empty($questionlist)) {
  272.         return array();

  274.  
  275.     $instances = get_records_sql("SELECT question,grade,id

  292. }

  312. /**

  316.  * @param float $rawgrade the unadjusted grade, fof example $attempt->sumgrades

  320. function quiz_rescale_grade($rawgrade, $quiz, $round = true) {
  321.     if ($quiz->sumgrades) {
  322.         $grade = $rawgrade * $quiz->grade / $quiz->sumgrades;
  323.         if ($round) {
  324.             $grade = round($grade, $quiz->decimalpoints);
  325.         }
  326.     } else {
  327.         $grade = 0;
  328.     }
  329.     return $grade;
  330. }

  369. /**
  370.  * The quiz grade is the score that student's results are marked out of. When it
  371.  * changes, the corresponding data in quiz_grades and quiz_feedback needs to be
  372.  * rescaled.
  373.  *
  374.  * @param float $newgrade the new maximum grade for the quiz.
  375.  * @param object $quiz the quiz we are updating. Passed by reference so its grade field can be updated too.
  376.  * @return boolean indicating success or failure.
  377.  */
  378. function quiz_set_grade($newgrade, &$quiz) {
  379.     // This is potentially expensive, so only do it if necessary.
  380.     if (abs($quiz->grade - $newgrade) < 1e-7) {
  381.         // Nothing to do.
  382.         return true;
  383.     }
  384.  
  385.     // Use a transaction, so that on those databases that support it, this is safer.
  386.     begin_sql();
  387.  
  388.     // Update the quiz table.
  389.     $success = set_field('quiz', 'grade', $newgrade, 'id', $quiz->instance);
  390.  
  391.     // Rescaling the other data is only possible if the old grade was non-zero.
  392.     if ($quiz->grade > 1e-7) {
  393.         global $CFG;
  394.  
  395.         $factor = $newgrade/$quiz->grade;
  396.         $quiz->grade = $newgrade;
  397.  
  398.         // Update the quiz_grades table.
  399.         $timemodified = time();
  400.         $success = $success && execute_sql("
  401.                 UPDATE {$CFG->prefix}quiz_grades
  402.                 SET grade = $factor * grade, timemodified = $timemodified
  403.                 WHERE quiz = $quiz->id
  404.         ", false);
  405.  
  406.         // Update the quiz_grades table.
  407.         $success = $success && execute_sql("
  408.                 UPDATE {$CFG->prefix}quiz_feedback
  409.                 SET mingrade = $factor * mingrade, maxgrade = $factor * maxgrade
  410.                 WHERE quizid = $quiz->id
  411.         ", false);
  412.     }
  413.

  426. /**

  429.  * @param object $quiz The quiz for which the best grade is to be calculated and then saved.
  430.  * @param integer $userid The userid to calculate the grade for. Defaults to the current user.
  431.  * @return boolean Indicates success or failure.
  432.  */
  433. function quiz_save_best_grade($quiz, $userid = null) {

  473. /**
  474.  * Calculate the overall grade for a quiz given a number of attempts by a particular user.
  475.  *
  476.  * @return float          The overall grade
  477.  * @param object $quiz    The quiz for which the best grade is to be calculated
  478.  * @param array $attempts An array of all the attempts of the user at the quiz
  479.  */
  480. function quiz_calculate_best_grade($quiz, $attempts) {
  481.  
  482.     switch ($quiz->grademethod) {
  483.  
  484.         case QUIZ_ATTEMPTFIRST:
  485.             foreach ($attempts as $attempt) {
  486.                 return $attempt->sumgrades;
  487.             }
  488.             break;
  489.  
  490.         case QUIZ_ATTEMPTLAST:
  491.             foreach ($attempts as $attempt) {
  492.                 $final = $attempt->sumgrades;
  493.             }
  494.             return $final;
  495.  
  496.         case QUIZ_GRADEAVERAGE:
  497.             $sum = 0;
  498.             $count = 0;
  499.             foreach ($attempts as $attempt) {
  500.                 $sum += $attempt->sumgrades;
  501.                 $count++;
  502.             }
  503.             return (float)$sum/$count;
  504.  
  505.         default:
  506.         case QUIZ_GRADEHIGHEST:
  507.             $max = 0;
  508.             foreach ($attempts as $attempt) {
  509.                 if ($attempt->sumgrades > $max) {
  510.                     $max = $attempt->sumgrades;
  511.                 }
  512.             }
  513.             return $max;
  514.     }
  515. }
  516.  
  517. /**
  518.  * Return the attempt with the best grade for a quiz
  519.  *
  520.  * Which attempt is the best depends on $quiz->grademethod. If the grade
  521.  * method is GRADEAVERAGE then this function simply returns the last attempt.
  522.  * @return object         The attempt with the best grade
  523.  * @param object $quiz    The quiz for which the best grade is to be calculated
  524.  * @param array $attempts An array of all the attempts of the user at the quiz
  525.  */
  526. function quiz_calculate_best_attempt($quiz, $attempts) {
  527.  
  528.     switch ($quiz->grademethod) {
  529.  
  530.         case QUIZ_ATTEMPTFIRST:
  531.             foreach ($attempts as $attempt) {
  532.                 return $attempt;
  533.             }
  534.             break;
  535.  
  536.         case QUIZ_GRADEAVERAGE: // need to do something with it :-)
  537.         case QUIZ_ATTEMPTLAST:
  538.             foreach ($attempts as $attempt) {
  539.                 $final = $attempt;
  540.             }
  541.             return $final;
  542.  
  543.         default:
  544.         case QUIZ_GRADEHIGHEST:
  545.             $max = -1;
  546.             foreach ($attempts as $attempt) {
  547.                 if ($attempt->sumgrades > $max) {
  548.                     $max = $attempt->sumgrades;
  549.                     $maxattempt = $attempt;
  550.                 }
  551.             }
  552.             return $maxattempt;
  553.     }
  554. }

  590. /**
  591.  * Upgrade states for an attempt to Moodle 1.5 model
  592.  *
  593.  * Any state that does not yet have its timestamp set to nonzero has not yet been upgraded from Moodle 1.4
  594.  * The reason these are still around is that for large sites it would have taken too long to
  595.  * upgrade all states at once. This function sets the timestamp field and creates an entry in the
  596.  * question_sessions table.
  597.  * @param object $attempt  The attempt whose states need upgrading
  598.  */
  599. function quiz_upgrade_states($attempt) {
  600.     global $CFG;
  601.     // The old quiz model only allowed a single response per quiz attempt so that there will be
  602.     // only one state record per question for this attempt.
  603.  
  604.     // We set the timestamp of all states to the timemodified field of the attempt.
  605.     execute_sql("UPDATE {$CFG->prefix}question_states SET timestamp = '$attempt->timemodified' WHERE attempt = '$attempt->uniqueid'", false);
  606.  
  607.     // For each state we create an entry in the question_sessions table, with both newest and
  608.     // newgraded pointing to this state.
  609.     // Actually we only do this for states whose question is actually listed in $attempt->layout.
  610.     // We do not do it for states associated to wrapped questions like for example the questions
  611.     // used by a RANDOM question
  612.     $session = new stdClass;
  613.     $session->attemptid = $attempt->uniqueid;
  614.     $questionlist = quiz_questions_in_quiz($attempt->layout);
  615.     if ($questionlist and $states = get_records_select('question_states', "attempt = '$attempt->uniqueid' AND question IN ($questionlist)")) {
  616.         foreach ($states as $state) {
  617.             $session->newgraded = $state->id;
  618.             $session->newest = $state->id;
  619.             $session->questionid = $state->question;
  620.             insert_record('question_sessions', $session, false);
  621.         }
  622.     }
  623. }

  678. /**
  679.  * Determine review options
  680.  *
  681.  * @param object $quiz the quiz instance.
  682.  * @param object $attempt the attempt in question.
  683.  * @param $context the roles and permissions context,
  684.  *           normally the context for the quiz module instance.
  685.  *
  686.  * @return object an object with boolean fields responses, scores, feedback,
  687.  *           correct_responses, solutions and general feedback
  688.  */
  689. function quiz_get_reviewoptions($quiz, $attempt, $context=null) {
  690.     $options = new stdClass;
  691.     $options->readonly = true;
  692.  
  693.     // Provide the links to the question review and comment script
  694.     $options->questionreviewlink = '/mod/quiz/reviewquestion.php';
  695.  
  696.     // Show a link to the comment box only for closed attempts
  697.     if ($attempt->timefinish && !is_null($context) && has_capability('mod/quiz:grade', $context)) {
  698.         $options->questioncommentlink = '/mod/quiz/comment.php';
  699.     }
  700.  
  701.     if (!is_null($context) && has_capability('mod/quiz:viewreports', $context) && 
  702.             has_capability('moodle/grade:viewhidden', $context) && !$attempt->preview) {
  703.         // People who can see reports and hidden grades should be shown everything,
  704.         // except during preview when teachers want to see what students see.
  705.         $options->responses = true;
  706.         $options->scores = true; 
  707.         $options->feedback = true;
  708.         $options->correct_responses = true;
  709.         $options->solutions = false;
  710.         $options->generalfeedback = true;
  711.         $options->overallfeedback = true;
  712.         $options->quizstate = QUIZ_STATE_TEACHERACCESS;
  713.     } else {
  714.         // Work out the state of the attempt ...
  715.         if (((time() - $attempt->timefinish) < 120) || $attempt->timefinish==0) {
  716.             $quiz_state_mask = QUIZ_REVIEW_IMMEDIATELY;
  717.             $options->quizstate = QUIZ_STATE_IMMEDIATELY;
  718.         } else if (!$quiz->timeclose or time() < $quiz->timeclose) {
  719.             $quiz_state_mask = QUIZ_REVIEW_OPEN;
  720.             $options->quizstate = QUIZ_STATE_OPEN;
  721.         } else {
  722.             $quiz_state_mask = QUIZ_REVIEW_CLOSED;
  723.             $options->quizstate = QUIZ_STATE_CLOSED;
  724.         }
  725.  
  726.         // ... and hence extract the appropriate review options. 
  727.         $options->responses = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_RESPONSES) ? 1 : 0;
  728.         $options->scores = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_SCORES) ? 1 : 0;
  729.         $options->feedback = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_FEEDBACK) ? 1 : 0;
  730.         $options->correct_responses = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_ANSWERS) ? 1 : 0;
  731.         $options->solutions = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_SOLUTIONS) ? 1 : 0;
  732.         $options->generalfeedback = ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_GENERALFEEDBACK) ? 1 : 0;
  733.         $options->overallfeedback = $attempt->timefinish && ($quiz->review & $quiz_state_mask & QUIZ_REVIEW_OVERALLFEEDBACK);
  734.     }
  735.  
  736.     return $options;
  737. }

  739. /**
  740.  * Combines the review options from a number of different quiz attempts.
  741.  * Returns an array of two ojects, so he suggested way of calling this
  742.  * funciton is:
  743.  * list($someoptions, $alloptions) = quiz_get_combined_reviewoptions(...)
  744.  *
  745.  * @param object $quiz the quiz instance.
  746.  * @param array $attempts an array of attempt objects.
  747.  * @param $context the roles and permissions context,
  748.  *           normally the context for the quiz module instance.
  749.  *
  750.  * @return array of two options objects, one showing which options are true for
  751.  *           at least one of the attempts, the other showing which options are true
  752.  *           for all attempts.
  753.  */
  754. function quiz_get_combined_reviewoptions($quiz, $attempts, $context=null) {
  755.     $fields = array('readonly', 'scores', 'feedback', 'correct_responses', 'solutions', 'generalfeedback', 'overallfeedback');
  756.     $someoptions = new stdClass;
  757.     $alloptions = new stdClass;
  758.     foreach ($fields as $field) {
  759.         $someoptions->$field = false;
  760.         $alloptions->$field = true;
  761.     }
  762.     foreach ($attempts as $attempt) {
  763.         $attemptoptions = quiz_get_reviewoptions($quiz, $attempt, $context);
  764.         foreach ($fields as $field) {
  765.             $someoptions->$field = $someoptions->$field || $attemptoptions->$field;
  766.             $alloptions->$field = $alloptions->$field && $attemptoptions->$field;
  767.         }
  768.     }
  769.     return array($someoptions, $alloptions);
  770. }
