  57.             // Store the chatroom details
  58.             $chatrooms[$cm->id] = $inst->name;
  59.         }

  61.         natcasesort($chatrooms);
  62.         
  63.     //--------------------------------------------------------

  69.         // Setup our default values
  70.         $sloodlemoduleid = (int)sloodle_get_value($settings, 'sloodlemoduleid', 0);

  74.     ///// GENERAL CONFIGURATION /////

  99.     }
