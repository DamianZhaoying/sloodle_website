  342.     /**
  343.     * Generates a random login security token.
  344.     * Uses mixed-case letters and numbers to generate a random 16-character string.
  345.     * @return string 
  346.     * @see sloodle_random_web_password()
  347.     */
  348.     function sloodle_random_security_token()
  349.     {
  350.         // Define the characters we can use in our token, and get the length of it
  351.         $str = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
  352.         $strlen = strlen($str) - 1;
  353.         // Prepare the token variable
  354.         $token = '';
  355.         // Loop once for each output character
  356.         for($length = 0; $length < 16; $length++) {
  357.             // Shuffle the string, then pick and store a random character
  358.             $str = str_shuffle($str);
  359.             $char = mt_rand(0, $strlen);
  360.             $token .= $str[$char];
  361.         }
